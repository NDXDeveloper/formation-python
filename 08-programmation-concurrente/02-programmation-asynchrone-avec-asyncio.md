# 8.2 : Programmation asynchrone avec asyncio

## Introduction

La programmation asynchrone avec `asyncio` permet de g√©rer des milliers de t√¢ches I/O-bound de fa√ßon efficace avec un seul thread. C'est parfait pour les serveurs web, les clients API, et les applications qui font beaucoup d'attente (r√©seau, fichiers).

### Analogie simple
Imaginez un **serveur de restaurant** :
- **Synchrone** : Le serveur prend une commande, va en cuisine, attend que le plat soit pr√™t, le sert, puis passe au client suivant
- **Asynchrone** : Le serveur prend une commande, la donne √† la cuisine, et pendant que le plat se pr√©pare, il prend d'autres commandes

## Concepts de base

### Coroutines et async/await

```python
import asyncio

def tache_synchrone(nom, duree):
```

### Gestion des erreurs asynchrones

```python
import asyncio
import random

async def tache_risquee(nom, taux_echec=0.3):
    """T√¢che qui peut √©chouer."""
    print(f"üé≤ {nom} d√©marr√©e (risque d'√©chec: {taux_echec*100}%)")

    await asyncio.sleep(random.uniform(0.5, 2))

    if random.random() < taux_echec:
        raise Exception(f"√âchec de {nom}")

    print(f"‚úÖ {nom} r√©ussie")
    return f"R√©sultat de {nom}"

async def executeur_robuste(nom, taux_echec=0.3, max_tentatives=3):
    """Ex√©cute une t√¢che avec retry automatique."""

    for tentative in range(1, max_tentatives + 1):
        try:
            print(f"üîÑ {nom} - Tentative {tentative}/{max_tentatives}")
            resultat = await tache_risquee(nom, taux_echec)
            return {"nom": nom, "resultat": resultat, "tentatives": tentative}

        except Exception as e:
            print(f"‚ùå {nom} - Tentative {tentative} √©chou√©e: {e}")

            if tentative == max_tentatives:
                return {"nom": nom, "erreur": str(e), "tentatives": tentative}

            # Attendre avant de retenter (backoff exponentiel)
            delai = 2 ** (tentative - 1)
            await asyncio.sleep(delai)

async def demo_gestion_erreurs():
    """D√©monstration de la gestion d'erreurs robuste."""

    taches = [
        executeur_robuste("API_1", taux_echec=0.2),
        executeur_robuste("API_2", taux_echec=0.4),
        executeur_robuste("API_3", taux_echec=0.6),
        executeur_robuste("API_4", taux_echec=0.1)
    ]

    print("üîß Test de robustesse avec retry automatique")
    resultats = await asyncio.gather(*taches, return_exceptions=True)

    print(f"\nüìä R√©sultats:")
    for resultat in resultats:
        if isinstance(resultat, dict):
            if "erreur" in resultat:
                print(f"  ‚ùå {resultat['nom']}: {resultat['erreur']} (apr√®s {resultat['tentatives']} tentatives)")
            else:
                print(f"  ‚úÖ {resultat['nom']}: succ√®s (tentative {resultat['tentatives']})")
        else:
            print(f"  üí• Erreur inattendue: {resultat}")

asyncio.run(demo_gestion_erreurs())
```

```python
async def client_http_concurrent(urls):
    """Client HTTP concurrent pour plusieurs URLs."""

    print(f"üåê T√©l√©chargement de {len(urls)} URLs en parall√®le")
    start_time = time.time()

    async with aiohttp.ClientSession() as session:
        # Lancer toutes les requ√™tes en parall√®le
        taches = [fetch_url(session, url) for url in urls]
        resultats = await asyncio.gather(*taches)

    duree = time.time() - start_time

    # Analyser les r√©sultats
    succes = sum(1 for r in resultats if r["success"])
    echecs = len(resultats) - succes

    print(f"  ‚úÖ Succ√®s: {succes}")
    print(f"  ‚ùå √âchecs: {echecs}")

        if resultat["success"]:
            print(f"  ‚úÖ {resultat['url']}: {resultat['status']} ({resultat['taille']} chars)")
        else:
            print(f"  ‚ùå {resultat['url']}: {resultat['erreur']}")

    return resultats
```

```python
            # Lire la requ√™te

                print(f"üì® {method} {path} (requ√™te #{self.stats['requetes']})")

                if path in self.routes:
                    response_body = await self.routes[path]()
                else:
                    response_body = f"<h1>404 - Page non trouv√©e</h1><p>Chemin: {path}</p>"

Content-Type: text/html; charset=utf-8\r
Content-Length: {len(response_body.encode('utf-8'))}\r
\r
{response_body}"""

                # Envoyer la r√©ponse
                writer.write(response.encode('utf-8'))
                await writer.drain()
```

### Exercice 2 : Gestionnaire de t√¢ches asynchrones

```python
import asyncio
import random
from enum import Enum

class StatutTache(Enum):
    EN_ATTENTE = "en_attente"
    EN_COURS = "en_cours"
    TERMINEE = "terminee"
    ECHOUEE = "echouee"

class Tache:
    """Repr√©sente une t√¢che asynchrone."""

    def __init__(self, nom, duree, taux_echec=0.1):
        self.nom = nom
        self.duree = duree
        self.taux_echec = taux_echec
        self.statut = StatutTache.EN_ATTENTE
        self.resultat = None
        self.erreur = None

    async def executer(self):
        """Ex√©cute la t√¢che."""
        self.statut = StatutTache.EN_COURS
        print(f"üîÑ D√©marrage: {self.nom}")

        try:
            # Simuler le travail
            await asyncio.sleep(self.duree)

            # Simuler un √©chec possible
            if random.random() < self.taux_echec:
                raise Exception(f"√âchec simul√© de {self.nom}")

            self.statut = StatutTache.TERMINEE
            self.resultat = f"R√©sultat de {self.nom}"
            print(f"‚úÖ Termin√©: {self.nom}")

        except Exception as e:
            self.statut = StatutTache.ECHOUEE
            self.erreur = str(e)
            print(f"‚ùå √âchec: {self.nom} - {e}")

class GestionnaireTaches:
    """Gestionnaire de t√¢ches asynchrones avec limite de concurrence."""

    def __init__(self, max_concurrent=3):
        self.max_concurrent = max_concurrent
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.taches = []

    def ajouter_tache(self, nom, duree, taux_echec=0.1):
        """Ajoute une t√¢che √† la liste."""
        tache = Tache(nom, duree, taux_echec)
        self.taches.append(tache)
        return tache

    async def executer_tache_limitee(self, tache):
        """Ex√©cute une t√¢che avec limite de concurrence."""
        async with self.semaphore:  # Limite le nombre de t√¢ches simultan√©es
            await tache.executer()

    async def executer_toutes(self):
        """Ex√©cute toutes les t√¢ches avec gestion de la concurrence."""
        if not self.taches:
            print("Aucune t√¢che √† ex√©cuter")
            return

        print(f"üöÄ Ex√©cution de {len(self.taches)} t√¢ches (max {self.max_concurrent} simultan√©es)")

        # Cr√©er les t√¢ches asyncio
        taches_async = [
            asyncio.create_task(self.executer_tache_limitee(tache))
            for tache in self.taches
        ]

        # Attendre toutes les t√¢ches
        await asyncio.gather(*taches_async, return_exceptions=True)

        # Afficher le r√©sum√©
        self.afficher_resume()

    def afficher_resume(self):
        """Affiche un r√©sum√© de l'ex√©cution."""
        stats = {
            StatutTache.TERMINEE: 0,
            StatutTache.ECHOUEE: 0
        }

        for tache in self.taches:
            stats[tache.statut] = stats.get(tache.statut, 0) + 1

        print(f"\nüìä R√©sum√©:")
        print(f"  ‚úÖ R√©ussites: {stats[StatutTache.TERMINEE]}")
        print(f"  ‚ùå √âchecs: {stats[StatutTache.ECHOUEE]}")
        print(f"  üìà Taux de r√©ussite: {stats[StatutTache.TERMINEE]/len(self.taches)*100:.1f}%")

# Test du gestionnaire

    # Ajouter des t√¢ches

    gestionnaire.ajouter_tache("Traitement_3", 3, 0.05)
    gestionnaire.ajouter_tache("Traitement_4", 1, 0.3)
    gestionnaire.ajouter_tache("Traitement_5", 2.5, 0.15)
    gestionnaire.ajouter_tache("Traitement_6", 1.8, 0.1)

    await gestionnaire.executer_toutes()

asyncio.run(demo_gestionnaire())
```

```python
import asyncio
import json
from datetime import datetime

class ChatServer:
    """Serveur de chat asynchrone simple."""

    def __init__(self, host='localhost', port=8889):
        self.host = host
        self.port = port
        self.clients = set()
        self.historique = []

    async def diffuser_message(self, message, expediteur=None):
        """Diffuse un message √† tous les clients connect√©s."""
        if not self.clients:
            return

        # Ajouter √† l'historique
        self.historique.append({
            "timestamp": datetime.now().isoformat(),
            "expediteur": expediteur,
            "message": message
        })

        # Garder seulement les 100 derniers messages
        self.historique = self.historique[-100:]

        # Diffuser aux clients connect√©s
        message_formate = json.dumps({
            "type": "message",
            "expediteur": expediteur,
            "message": message,
            "timestamp": datetime.now().isoformat()
        })

        # Envoyer √† tous les clients (en parall√®le)
        if self.clients:
            await asyncio.gather(
                *[self.envoyer_message(client, message_formate) for client in self.clients],
                return_exceptions=True
            )

    async def envoyer_message(self, writer, message):
        """Envoie un message √† un client sp√©cifique."""
        try:
            writer.write(f"{message}\n".encode())
            await writer.drain()
        except Exception:
            # Client d√©connect√©, l'enlever de la liste
            self.clients.discard(writer)

    async def gerer_client(self, reader, writer):
        """G√®re un client connect√©."""
        client_addr = writer.get_extra_info('peername')
        print(f"üë§ Nouveau client connect√©: {client_addr}")

        self.clients.add(writer)

        # Envoyer l'historique au nouveau client
        if self.historique:
            historique_json = json.dumps({
                "type": "historique",
                "messages": self.historique[-10:]  # 10 derniers messages
            })
            await self.envoyer_message(writer, historique_json)

        # Annoncer l'arriv√©e
        await self.diffuser_message(f"Un nouvel utilisateur s'est connect√©", "Syst√®me")

        try:
            while True:
                # Lire les messages du client
                data = await reader.readline()
                if not data:
                    break

                message = data.decode().strip()
                if message:
                    print(f"üí¨ {client_addr}: {message}")
                    await self.diffuser_message(message, str(client_addr))

        except Exception as e:
            print(f"‚ùå Erreur avec {client_addr}: {e}")

        finally:
            # Nettoyer lors de la d√©connexion
            self.clients.discard(writer)
            writer.close()
            await writer.wait_closed()

            await self.diffuser_message(f"Un utilisateur s'est d√©connect√©", "Syst√®me")
            print(f"üëã Client d√©connect√©: {client_addr}")

    async def demarrer(self):
        """D√©marre le serveur de chat."""
        server = await asyncio.start_server(
            self.gerer_client,
            self.host,
            self.port
```

```python
import asyncio
import random

class ProducteurConsommateur:
    """Pattern Producer-Consumer avec asyncio."""

    def __init__(self, taille_queue=5):
        self.queue = asyncio.Queue(maxsize=taille_queue)
        self.actif = True

    async def producteur(self, nom, nb_items):
        """Produit des √©l√©ments de fa√ßon asynchrone."""
        for i in range(nb_items):
            if not self.actif:
                break

            item = f"{nom}_item_{i+1}"

            # Simuler la production
            await asyncio.sleep(random.uniform(0.1, 0.5))

            await self.queue.put(item)
            print(f"üì¶ {nom} a produit: {item}")

        print(f"‚úÖ {nom} termin√©")

    async def consommateur(self, nom):
        """Consomme des √©l√©ments de fa√ßon asynchrone."""
        while self.actif:
            try:
                # Attendre un √©l√©ment avec timeout
                item = await asyncio.wait_for(self.queue.get(), timeout=2.0)

                print(f"üîÑ {nom} traite: {item}")

                # Simuler le traitement
                await asyncio.sleep(random.uniform(0.2, 0.8))

                self.queue.task_done()
                print(f"‚úÖ {nom} a termin√©: {item}")

            except asyncio.TimeoutError:
                print(f"‚è∞ {nom}: timeout, v√©rification d'arr√™t")
                if self.queue.empty():
                    break

    async def executer(self, duree=10):
        """Ex√©cute le syst√®me producer-consumer."""
        print(f"üè≠ D√©marrage du syst√®me (dur√©e: {duree}s)")

        # Cr√©er les t√¢ches
        taches = [
            asyncio.create_task(self.producteur("Prod1", 5)),
            asyncio.create_task(self.producteur("Prod2", 4)),
            asyncio.create_task(self.consommateur("Cons1")),
            asyncio.create_task(self.consommateur("Cons2"))
        ]

        # Arr√™ter apr√®s la dur√©e sp√©cifi√©e
        await asyncio.sleep(duree)
        self.actif = False

        # Attendre que toutes les t√¢ches se terminent
        await asyncio.gather(*taches, return_exceptions=True)

        print("üéâ Syst√®me arr√™t√©")
```
