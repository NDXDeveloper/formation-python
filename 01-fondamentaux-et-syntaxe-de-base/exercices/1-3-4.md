    """
    Analyse la casse du texte (majuscules/minuscules).

    Args:
        texte (str): Le texte Ã  analyser

    Returns:
        dict: Statistiques de casse
    """
    majuscules = 0

    for caractere in texte:

            majuscules += 1
        elif caractere.islower():
            minuscules += 1

    return {

        'minuscules': minuscules

    return frequences

def mode_recherche():
    """Mode recherche dans le texte."""
    print("\nðŸ” === RECHERCHE DANS LE TEXTE ===")

    texte = input("ðŸ“ Entrez votre texte : ")
    if not texte.strip():
        print("âŒ Aucun texte saisi")
        return

    while True:
        print(f"\nTexte : \"{texte}\"")
        terme = input("ðŸ” Terme Ã  rechercher (ou 'fin' pour arrÃªter) : ")

        if terme.lower() == 'fin':
            break

        if not terme:
            print("âŒ Terme vide")
            continue

        # Recherche case-sensitive
        occurrences = texte.count(terme)

        # Recherche case-insensitive
        occurrences_insensitive = texte.lower().count(terme.lower())

        print(f"ðŸ“Š RÃ©sultats de recherche :")
        print(f"   â€¢ Occurrences exactes : {occurrences}")
        print(f"   â€¢ Occurrences (ignore casse) : {occurrences_insensitive}")

        if occurrences_insensitive > 0:
            # Trouver les positions
            positions = []
            texte_lower = texte.lower()
            terme_lower = terme.lower()
            start = 0

            while True:
                pos = texte_lower.find(terme_lower, start)
                if pos == -1:
                    break
                positions.append(pos)
                start = pos + 1

            print(f"   â€¢ Positions : {positions}")

            # Afficher le contexte
            if len(positions) <= 5:  # Limiter l'affichage
                print(f"   â€¢ Contexte :")
                for pos in positions:
                    debut = max(0, pos - 10)
                    fin = min(len(texte), pos + len(terme) + 10)
                    contexte = texte[debut:fin]
                    print(f"     ...{contexte}...")
